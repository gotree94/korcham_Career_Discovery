## 실습 1: 첫 번째 센서 읽기 (초음파)

**난이도**: ⭐
**소요시간**: 30분
**목표**: 초음파 센서 1개를 연결하고 거리값을 읽어본다.

### Step 1: 연결
- HC-SR04의 VCC → Jetson 5V (Pin 2)
- HC-SR04의 GND → Jetson GND (Pin 6)
- HC-SR04의 Trig → GPIO17 (Pin 11)
- HC-SR04의 Echo → GPIO18 (Pin 12) (저항 분배 필수!)

### Step 2: 코드 작성
```bash
nano ~/ultrasonic_test.py
```

```python
import Jetson.GPIO as GPIO
import time

TRIG = 17
ECHO = 18

GPIO.setmode(GPIO.BCM)
GPIO.setup(TRIG, GPIO.OUT)
GPIO.setup(ECHO, GPIO.IN)

try:
    while True:
        GPIO.output(TRIG, True)
        time.sleep(0.00001)
        GPIO.output(TRIG, False)
        
        while GPIO.input(ECHO) == 0:
            start = time.time()
        while GPIO.input(ECHO) == 1:
            end = time.time()
        
        dist = (end - start) * 34300 / 2
        print(f"거리: {dist:.1f} cm")
        
        if dist < 10:
            print("⚠️ 너무 가까워요!")
        elif dist < 30:
            print("⚠️ 조심!")
        else:
            print("✅ 안전")
        
        time.sleep(0.3)

except KeyboardInterrupt:
    GPIO.cleanup()
    print("\n종료")
```

### Step 3: 실행
```bash
python3 ~/ultrasonic_test.py
```

## 실습 2: 3개 초음파 센서 동시 읽기

**난이도**: ⭐⭐
**소요시간**: 30분
**목표**: 3개 초음파 센서로 전방 3방향(좌/중/우) 거리를 동시에 측정한다.

### 연결
```
초음파 1 (좌): Trig=GPIO17, Echo=GPIO18
초음파 2 (중): Trig=GPIO27, Echo=GPIO22
초음파 3 (우): Trig=GPIO5,  Echo=GPIO6
```

### 코드
```python
import Jetson.GPIO as GPIO
import time

SENSORS = {
    'LEFT':  {'trig': 17, 'echo': 18},
    'CENTER': {'trig': 27, 'echo': 22},
    'RIGHT': {'trig': 5, 'echo': 6},
}

GPIO.setmode(GPIO.BCM)
for name, pin in SENSORS.items():
    GPIO.setup(pin['trig'], GPIO.OUT)
    GPIO.setup(pin['echo'], GPIO.IN)

def measure(name):
    t = SENSORS[name]['trig']
    e = SENSORS[name]['echo']
    
    GPIO.output(t, True)
    time.sleep(0.00001)
    GPIO.output(t, False)
    
    timeout = time.time() + 0.1
    while GPIO.input(e) == 0 and time.time() < timeout:
        start = time.time()
    while GPIO.input(e) == 1 and time.time() < timeout:
        end = time.time()
    
    return (end - start) * 34300 / 2

try:
    while True:
        d_left = measure('LEFT')
        d_center = measure('CENTER')
        d_right = measure('RIGHT')
        
        print(f"좌:{d_left:6.1f}cm | 중:{d_center:6.1f}cm | 우:{d_right:6.1f}cm")
        
        # 간단한 장애물 방향 판단
        if d_center < 20:
            print("  → 정면 장애물! 회전 필요")
        elif d_left < 20:
            print("  → 좌측 장애물! 우회")
        elif d_right < 20:
            print("  → 우측 장애물! 좌회")
        else:
            print("  → 전방 이상 없음")
        
        time.sleep(0.2)

except KeyboardInterrupt:
    GPIO.cleanup()
```

**도전과제**: 측정 간격을 줄여서 더 부드러운 값을 만들어보세요.

## 실습 5: LCD에 센서값 출력

**난이도**: ⭐⭐
**소요시간**: 30분
**목표**: I2C LCD에 지금까지 배운 모든 센서값을 출력한다.

```bash
# I2C 주소 확인
sudo i2cdetect -y -r 1
# 0x27 또는 0x3F 확인
```

```python
import smbus2
import time

class LCD:
    def __init__(self, addr=0x27):
        self.bus = smbus2.SMBus(1)
        self.addr = addr
        self.init()
    
    def init(self):
        # LCD 초기화 명령
        self.write_cmd(0x33)
        self.write_cmd(0x32)
        self.write_cmd(0x28)  # 4비트, 2라인
        self.write_cmd(0x0C)  # 디스플레이 온, 커서 오프
        self.write_cmd(0x06)  # 커서 증가
        self.write_cmd(0x01)  # 클리어
    
    def write_cmd(self, cmd):
        self.bus.write_byte_data(self.addr, 0x80, cmd)
        time.sleep(0.005)
    
    def write_text(self, text, line=0):
        addr = 0x80 + (0x40 if line else 0x00)
        self.write_cmd(addr)
        for c in text.ljust(16)[:16]:
            self.bus.write_byte_data(self.addr, 0x40, ord(c))
            time.sleep(0.002)
    
    def clear(self):
        self.write_cmd(0x01)
        time.sleep(0.05)

lcd = LCD(0x27)
lcd.write_text("AGX Rover Ready!", 0)
lcd.write_text("Team Awesome", 1)
```

**도전과제**: 초음파 + IR + 습도 데이터를 LCD에 0.5초마다 갱신

## 실습 6: 옴니휠 1개 돌려보기

**난이도**: ⭐⭐
**소요시간**: 30분
**목표**: 모터 드라이버를 통해 옴니휠 1개를 정/역회전시킨다.

```python
import Jetson.GPIO as GPIO
import time

# M1 모터
IN1 = 5
IN2 = 6
PWM_PIN = 12

GPIO.setmode(GPIO.BCM)
GPIO.setup(IN1, GPIO.OUT)
GPIO.setup(IN2, GPIO.OUT)
GPIO.setup(PWM_PIN, GPIO.OUT)

pwm = GPIO.PWM(PWM_PIN, 1000)  # 1kHz
pwm.start(0)

def motor_forward(speed=50):
    GPIO.output(IN1, True)
    GPIO.output(IN2, False)
    pwm.ChangeDutyCycle(speed)

def motor_backward(speed=50):
    GPIO.output(IN1, False)
    GPIO.output(IN2, True)
    pwm.ChangeDutyCycle(speed)

def motor_stop():
    GPIO.output(IN1, False)
    GPIO.output(IN2, False)
    pwm.ChangeDutyCycle(0)

# 테스트
print("전진 3초")
motor_forward(70)
time.sleep(3)

print("정지 1초")
motor_stop()
time.sleep(1)

print("후진 3초")
motor_backward(70)
time.sleep(3)

print("정지")
motor_stop()
GPIO.cleanup()
```

## 실습 7: 3개 옴니휠 동기 제어

**난이도**: ⭐⭐⭐
**소요시간**: 1h
**목표**: 3개 옴니휠을 동시에 제어하여 전/후/좌/우/회전을 구현한다.

```python
import Jetson.GPIO as GPIO
import math
import time
import sys

# 모터 핀 설정
MOTORS = {
    'M1': {'in1': 5, 'in2': 6, 'pwm': 12},
    'M2': {'in1': 13, 'in2': 19, 'pwm': 33},
    'M3': {'in1': 20, 'in2': 21, 'pwm': 32},
}

# GPIO 설정
GPIO.setmode(GPIO.BCM)
pwms = {}
for name, pins in MOTORS.items():
    GPIO.setup(pins['in1'], GPIO.OUT)
    GPIO.setup(pins['in2'], GPIO.OUT)
    GPIO.setup(pins['pwm'], GPIO.OUT)
    pwms[name] = GPIO.PWM(pins['pwm'], 1000)
    pwms[name].start(0)

def set_motor(name, speed):
    # speed: -100 ~ 100
    pins = MOTORS[name]
    if speed > 5:
        GPIO.output(pins['in1'], True)
        GPIO.output(pins['in2'], False)
    elif speed < -5:
        GPIO.output(pins['in1'], False)
        GPIO.output(pins['in2'], True)
    else:
        GPIO.output(pins['in1'], False)
        GPIO.output(pins['in2'], False)
        speed = 0
    pwms[name].ChangeDutyCycle(min(abs(speed), 100))

def omni_drive(vx, vy, omega):
    R = 1.0
    sqrt3_half = math.sqrt(3) / 2
    
    v1 = vy * sqrt3_half - vx * 0.5 + R * omega
    v2 = -vy * sqrt3_half - vx * 0.5 + R * omega
    v3 = vx + R * omega
    
    max_v = max(abs(v1), abs(v2), abs(v3), 1)
    set_motor('M1', (v1 / max_v) * 100)
    set_motor('M2', (v2 / max_v) * 100)
    set_motor('M3', (v3 / max_v) * 100)

# 키보드 제어
def keyboard_control():
    import tty, termios
    fd = sys.stdin.fileno()
    old = termios.tcgetattr(fd)
    tty.setraw(fd)
    
    print("WASD: 이동  | Q/E: 회전  | SPACE: 정지  | X: 종료")
    try:
        while True:
            ch = sys.stdin.read(1)
            if ch == 'w': omni_drive(0, 50, 0)
            elif ch == 's': omni_drive(0, -50, 0)
            elif ch == 'a': omni_drive(-50, 0, 0)
            elif ch == 'd': omni_drive(50, 0, 0)
            elif ch == 'q': omni_drive(0, 0, 30)
            elif ch == 'e': omni_drive(0, 0, -30)
            elif ch == ' ': omni_drive(0, 0, 0)
            elif ch == 'x': break
            time.sleep(0.05)
    finally:
        termios.tcsetattr(fd, termios.TCSADRAIN, old)
        omni_drive(0, 0, 0)
        for p in pwms.values(): p.stop()
        GPIO.cleanup()

keyboard_control()
```

> **참고**: vx, vy, omega 값과 바퀴 방향이 실제와 다르면 모터 방향이나 공식을 조정

## 실습 8: ROS2 통신으로 센서-모터 연결

**난이도**: ⭐⭐⭐
**소요시간**: 1h
**목표**: 센서 데이터를 ROS2 토픽으로 발행하고, 모터 제어 노드가 구독하여 장애물 회피를 구현한다.

### publisher (센서 노드)
```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from std_msgs.msg import Float32MultiArray
import Jetson.GPIO as GPIO
import time

class SensorNode(Node):
    def __init__(self):
        super().__init__('sensor_node')
        self.pub = self.create_publisher(Float32MultiArray, 'sensor_data', 10)
        self.timer = self.create_timer(0.2, self.publish_sensors)
        
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(17, GPIO.OUT); GPIO.setup(18, GPIO.IN)
    
    def publish_sensors(self):
        msg = Float32MultiArray()
        # [초음파거리, IR1, IR2, IR3]
        msg.data = [self.measure_ultrasonic(), 0.0, 0.0, 0.0]
        self.pub.publish(msg)
    
    def measure_ultrasonic(self):
        GPIO.output(17, True)
        time.sleep(0.00001)
        GPIO.output(17, False)
        while GPIO.input(18) == 0: start = time.time()
        while GPIO.input(18) == 1: end = time.time()
        return (end - start) * 34300 / 2

def main():
    rclpy.init()
    node = SensorNode()
    rclpy.spin(node)
    GPIO.cleanup()
```

## 실습 9: 벽 따라가기 (Wall Following)

**난이도**: ⭐⭐⭐⭐
**소요시간**: 1.5h
**목표**: LiDAR 데이터로 벽과의 거리를 측정하고, 일정 거리를 유지하며 벽을 따라 주행한다.

```python
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import LaserScan
from geometry_msgs.msg import Twist

class WallFollower(Node):
    def __init__(self):
        super().__init__('wall_follower')
        self.sub = self.create_subscription(LaserScan, '/scan', self.scan_callback, 10)
        self.pub = self.create_publisher(Twist, '/cmd_vel', 10)
        
        self.target_dist = 0.5  # 목표 거리 (m)
        self.kp = 1.0  # P 게인
    
    def scan_callback(self, msg):
        # 오른쪽 90도 방향 거리
        right_idx = len(msg.ranges) // 4
        right_dist = min(msg.ranges[right_idx-10:right_idx+10])
        
        # 전방 거리
        front_dist = min(msg.ranges[0:10] + msg.ranges[-10:])
        
        twist = Twist()
        
        if front_dist < 0.4:
            # 전방 장애물 → 좌회전
            twist.angular.z = 1.0
            twist.linear.x = 0.0
        else:
            # 벽 따라가기 (PD)
            error = self.target_dist - right_dist
            twist.angular.z = -error * self.kp
            twist.linear.x = 0.3
        
        self.pub.publish(twist)

def main():
    rclpy.init()
    node = WallFollower()
    rclpy.spin(node)
```

**도전과제**: P 게인 값을 바꿔보며 주행 안정성 변화 관찰
