## 3. 센서별 연결 및 코드

### 3.1 초음파 센서 HC-SR04

**연결**
| HC-SR04 | Jetson GPIO |
|---------|-------------|
| VCC | 5V (Pin 2 또는 4) |
| GND | GND |
| Trig | GPIO (Pin 11) |
| Echo | GPIO (Pin 12) - **반드시 저항 분배(5V→3.3V)** |

**Python 코드 예시**
```python
import Jetson.GPIO as GPIO
import time

TRIG = 17
ECHO = 18

def setup():
    GPIO.setmode(GPIO.BCM)
    GPIO.setup(TRIG, GPIO.OUT)
    GPIO.setup(ECHO, GPIO.IN)

def measure_distance():
    GPIO.output(TRIG, True)
    time.sleep(0.00001)
    GPIO.output(TRIG, False)
    
    pulse_start = time.time()
    pulse_end = time.time()
    
    while GPIO.input(ECHO) == 0:
        pulse_start = time.time()
    while GPIO.input(ECHO) == 1:
        pulse_end = time.time()
    
    pulse_duration = pulse_end - pulse_start
    distance = pulse_duration * 34300 / 2
    return distance

if __name__ == '__main__':
    setup()
    try:
        while True:
            dist = measure_distance()
            print(f"Distance: {dist:.1f} cm")
            time.sleep(0.5)
    except KeyboardInterrupt:
        GPIO.cleanup()
```

### 4.2 옴니휠 3개 운동학

```
각 바퀴 120° 배치

v1 = -0.5*Vx + (√3/2)*Vy + R*ω
v2 = -0.5*Vx - (√3/2)*Vy + R*ω
v3 = Vx + R*ω

v1, v2, v3: 각 바퀴 속도
Vx, Vy: 로봇 전진/후진 속도
ω: 회전 각속도
R: 로봇 중심에서 바퀴까지 거리
```

```python
import math
import Jetson.GPIO as GPIO

# 모터 핀 설정
MOTOR_PINS = {
    'M1': {'in1': 5, 'in2': 6, 'pwm': 12},
    'M2': {'in1': 13, 'in2': 19, 'pwm': 33},
    'M3': {'in1': 20, 'in2': 21, 'pwm': 32},
}

def set_motor_speed(motor, speed):
    # speed: -100 ~ 100
    in1, in2, pwm = MOTOR_PINS[motor].values()
    
    if speed > 0:
        GPIO.output(in1, True)
        GPIO.output(in2, False)
    elif speed < 0:
        GPIO.output(in1, False)
        GPIO.output(in2, True)
    else:
        GPIO.output(in1, False)
        GPIO.output(in2, False)

    pwm_value = int(abs(speed) / 100 * 255)
    # PWM 출력 (하드웨어 PWM 또는 소프트웨어 PWM)

def omni_drive(vx, vy, omega):
    R = 0.15  # 로봇 반경 (m)
    sqrt3 = math.sqrt(3)
    
    v1 = -0.5*vx + (sqrt3/2)*vy + R*omega
    v2 = -0.5*vx - (sqrt3/2)*vy + R*omega
    v3 = vx + R*omega
    
    # -100~100 범위로 정규화
    max_v = max(abs(v1), abs(v2), abs(v3), 1)
    v1 = (v1 / max_v) * 100
    v2 = (v2 / max_v) * 100
    v3 = (v3 / max_v) * 100
    
    set_motor_speed('M1', v1)
    set_motor_speed('M2', v2)
    set_motor_speed('M3', v3)
```

## 5. LCD (I2C 1602 / 2004)

```python
import smbus2
import time

# I2C 주소 확인 (보통 0x27 또는 0x3F)
# sudo i2cdetect -y -r 1

class LCD:
    def __init__(self, addr=0x27):
        self.bus = smbus2.SMBus(1)
        self.addr = addr
        self.init_lcd()
    
    def write_text(self, text, line=0):
        # line 0: 첫째줄, line 1: 둘째줄
        self.set_cursor(line, 0)
        text = text.ljust(16)[:16]
        for char in text:
            self.write_char(ord(char))
    
    def display_sensor_data(self, dist, ir, temp, humid, flame_status):
        self.write_text(f"D:{dist:.0f}cm IR:{ir}", 0)
        self.write_text(f"F:{flame_status} T:{temp}C", 1)

# 사용 예
lcd = LCD(0x27)
lcd.display_sensor_data(45, 1, 25.5, 60, "OK")
```

## 7. 문제 해결 (Troubleshooting)

| 증상 | 원인 | 해결 |
|------|------|------|
| 초음파 값 0 or 튐 | Echo 핀 전압 불일치 | 5V→3.3V 레벨시프터 추가 |
| IR 센서 값 항상 0/1 | VCC/GND 잘못 연결 | 핀맵 재확인 |
| LiDAR 미동작 | USB 권한 | `sudo chmod 666 /dev/ttyUSB0` |
| I2C LCD 안나옴 | I2C 주소 틀림 | `i2cdetect -y -r 1`로 주소 확인 |
| 모터 안돌아감 | PWM 핀 비활성 | `sudo jetson_clocks` 실행 |
| ROS2 노드 실행 안됨 | setup.bash 미실행 | `source install/setup.bash` |
